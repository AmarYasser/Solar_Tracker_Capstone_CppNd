1. Install dependencies

	sudo apt install build-essential cmake git pkg-config libgtk-3-dev \
	    libavcodec-dev libavformat-dev libswscale-dev libv4l-dev \
	    libxvidcore-dev libx264-dev libjpeg-dev libpng-dev libtiff-dev \
	    gfortran openexr libatlas-base-dev python3-dev python3-numpy \
	    libtbb2 libtbb-dev libdc1394-22-dev

2. Clone opencv and contrib repositories

	mkdir ~/opencv_build && cd ~/opencv_build
	git clone https://github.com/opencv/opencv.git
	git clone https://github.com/opencv/opencv_contrib.git


3. Create temporary build directory

	cd ~/opencv_build/opencv
	mkdir build
	cd build


4. Setup opencv

	cmake -D CMAKE_BUILD_TYPE=RELEASE \
	    -D CMAKE_INSTALL_PREFIX=/usr/local \
	    -D INSTALL_C_EXAMPLES=ON \
	    -D INSTALL_PYTHON_EXAMPLES=ON \
	    -D OPENCV_GENERATE_PKGCONFIG=ON \
	    -D OPENCV_EXTRA_MODULES_PATH=~/opencv_build/opencv_contrib/modules \
	    -D BUILD_EXAMPLES=ON ..


5. compilation process

	make -j{number of processors' cores}


6. sudo make install

7. Update c++ configuration file in VScode with the lib path inside include/Opencv4 

	{
	    "configurations": [
		{
		    "name": "Linux",
		    "includePath": [
		        "${workspaceFolder}/**",
		        "/usr/local/include/opencv4/**"
		    ],
		    "defines": [],
		    "compilerPath": "/usr/bin/gcc",
		    "cStandard": "gnu11",
		    "cppStandard": "c++17",
		    "intelliSenseMode": "clang-x64"
		}
	    ],
	    "version": 4
	}


7. import opencv package inside *Makefiles*

	OPENCV = `pkg-config opencv4 --cflags --libs`
